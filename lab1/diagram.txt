@startuml
title Data model for dictionary app with learning progress

entity User {
  *id: int <<PK>>
  --
  name: text
  email: text
}

entity Dictionary {
  *id: int <<PK>>
  --
  title: text
  description: text
  user_id: int <<FK>>
}

entity Word {
  *id: int <<PK>>
  --
  key: text
  value: text
}

entity Progress {
  *pair_id: int <<FK, PK>>
  --
  knowledge_level: int
  repetitions: int
  correct_in_a_row: int
  last_repetition: timestamp
  next_repetition: timestamp
}

entity Pair {
  *id: int <<PK>>
  *dictionary_id: int <<FK>>
  *word_id: int <<FK>>
  --
}

' Relationships
User ||--o{ Dictionary : creates
Pair ||-|| Progress : tracks
Word ||--o{ Pair : appears in
Dictionary ||-o{ Pair : contains
@enduml
