@startuml
title Модель данных для приложения-словаря с прогрессом изучения

entity Пользователь {
  *id: int <<PK>>
  --
  имя: text
  email: text
}

entity Словари {
  *id: int <<PK>>
  --
  название: text
  описание: text
  пользователь_id: int <<FK>>
}

entity Слова {
  *id: int <<PK>>
  --
  ключ: text
  значение: text
}

entity Прогресс {
  *пользователь_id: int <<FK>>
  *пара_id: int <<FK>>
  --
  уровень_знания: int      "0..5 (0 – не изучено, 5 – идеально)"
  повторений: int
  правильных_подряд: int
  последнее_повторение: timestamp
  следующее_повторение: timestamp
}

entity Пара {
  *id: int <<FK>>
  *словарь_id: int <<FK>>
  *слово_id: int <<FK>>
  --
}

' Связи
Пользователь ||--o{ Словари : создаёт
Пара ||-|| Прогресс : отслеживается
Слова ||-|| Пара : создают
Словари ||-o{ Пара: создают
@enduml
